\begin{proposition}{First Isomorphism Theorem}{}
  If $f \from G \surjto G'$, then
  \begin{itemize}
    \item $\ker f \normalto G$.
    \item $H \iso G / \ker f$.
    \item $\im f \le H$.
  \end{itemize}
\end{proposition}
